import pandas as pd
import plotly.express as px
import plotly.graph_objects as go
from collections import Counter
import re

# Load data
sb_locs  = pd.read_csv("starbucks-locations.csv")
sb_nutr  = pd.read_csv("starbucks-menu-nutrition.csv", skipinitialspace=True)
usa_pop  = pd.read_csv("us_state_pop.csv")
usa_states = pd.read_csv("states.csv")

# Filter US locations and count stores per state
sb_usa = sb_locs[sb_locs["Country"] == "US"]
sb_locs_state = (
    sb_usa
    .groupby("State/Province")
    .size()
    .reset_index(name="n_stores")
    .rename(columns={"State/Province": "state"})
)

# Merge with population (via state abbreviation)
usa_pop_abbr = usa_pop.merge(usa_states, left_on="state", right_on="State")
sb_locs_state = sb_locs_state.merge(
    usa_pop_abbr, left_on="state", right_on="Abbreviation"
)

# Top 10 words in menu item names
stop_words = {"the", "and", "with", "a", "an", "of", "in", "or", "no", "on"}
word_counts = Counter(
    w for name in sb_nutr["Item"].dropna()
    for w in re.findall(r"[a-zA-Z]+", name.lower())
    if len(w) > 2 and w not in stop_words
)
top10_words = [w for w, _ in word_counts.most_common(10)]

def find_top_word(item_name):
    for w in top10_words:
        if w in str(item_name).lower():
            return w
    return None

sb_nutr["top_word"] = sb_nutr["Item"].apply(find_top_word)
sb_top = sb_nutr.dropna(subset=["top_word"]).copy()
